52class BaseAuthPlugin(object): 

53 """Base class for authentication plugins 

54 

55 

56 Classes inheriting from BaseAuthPlugin should implement the method 

57 prepare_password(). When instantiating, auth_data argument is 

58 required. The username, password and database are optional. The 

59 ssl_enabled argument can be used to tell the plugin whether SSL is 

60 active or not. 

61 

62 The method auth_response() method is used to retrieve the password 

63 which was prepared by prepare_password(). 

64 """ 

65 

66 requires_ssl = False 

67 plugin_name = '' 

68 

69 def __init__(self, auth_data, username=None, password=None, database=None, 

70 ssl_enabled=False, instance=None): 

71 """Initialization""" 

72 self._auth_data = auth_data 

73 self._username = username 

74 self._password = password 

75 self._database = database 

76 self._ssl_enabled = ssl_enabled 

77 

78 def prepare_password(self): 

79 """Prepares and returns password to be send to MySQL 

80 

81 This method needs to be implemented by classes inheriting from 

82 this class. It is used by the auth_response() method. 

83 

84 Raises NotImplementedError. 

85 """ 

86 raise NotImplementedError

88 def auth_response(self): 

89 """Returns the prepared password to send to MySQL 

90 

91 Raises InterfaceError on errors. For example, when SSL is required 

92 by not enabled. 

93 

94 Returns str 

95 """ 

96 if self.requires_ssl and not self._ssl_enabled: 

97 raise errors.InterfaceError("{name} requires SSL".format( 

98 name=self.plugin_name))

108 def prepare_password(self): 

109 """Prepares and returns password as native MySQL 4.1+ password""" 

110 if not self._auth_data: 

111 raise errors.InterfaceError("Missing authentication data (seed)") 

112 

113 if not self._password: 

114 return b'' 

115 password = self._password 

116 

117 if isinstance(self._password, str): 

118 password = self._password.encode('utf-8') 

119 else: 

120 password = self._password 

121 

122 auth_data = self._auth_data 

123 

124 hash4 = None 

125 try: 

126 hash1 = sha1(password).digest() 

127 hash2 = sha1(hash1).digest() 

128 hash3 = sha1(auth_data + hash2).digest() 

129 xored = [h1 ^ h3 for (h1, h3) in zip(hash1, hash3)] 

130 hash4 = struct.pack('20B', *xored) 

131 except Exception as exc: 

132 raise errors.InterfaceError( 

133 "Failed scrambling password; {0}".format(exc)) 

134 

135 return hash4

189 def _scramble(self): 

190 """ Returns a scramble of the password using a Nonce sent by the 

191 server. 

192 

193 The scramble is of the form: 

194 XOR(SHA2(password), SHA2(SHA2(SHA2(password)), Nonce)) 

195 """ 

196 if not self._auth_data: 

197 raise errors.InterfaceError("Missing authentication data (seed)") 

198 

199 if not self._password: 

200 return b'' 

201 

202 password = self._password.encode('utf-8') \ 

203 if isinstance(self._password, str) else self._password 

204 auth_data = self._auth_data 

205 

206 hash1 = sha256(password).digest() 

207 hash2 = sha256() 

208 hash2.update(sha256(hash1).digest()) 

209 hash2.update(auth_data) 

210 hash2 = hash2.digest() 

211 xored = [h1 ^ h2 for (h1, h2) in zip(hash1, hash2)] 

212 hash3 = struct.pack('32B', *xored) 

213 

214 return hash3

268 def _xor(self, bytes1, bytes2): 

269 return bytes([b1 ^ b2 for b1, b2 in zip(bytes1, bytes2)])

271 def _hmac(self, password, salt): 

272 digest_maker = hmac.new(password, salt, self.def_digest_mode)

275 def _hi(self, password, salt, count): 

276 """Prepares Hi 

277 Hi(password, salt, iterations) where Hi(p,s,i) is defined as 

278 PBKDF2 (HMAC, p, s, i, output length of H). 

279 

280 """ 

281 pw = password.encode() 

282 hi = self._hmac(pw, salt + b'\x00\x00\x00\x01') 

283 aux = hi 

284 for _ in range(count - 1): 

285 aux = self._hmac(pw, aux) 

286 hi = self._xor(hi, aux) 

287 return hi

410 def auth_accept_close_handshake(self, message): 

411 """Accept handshake and generate closing handshake message for server. 

412 

413 This method verifies the server authenticity from the given message 

414 and included signature and generates the closing handshake for the 

415 server. 

416 

417 When this method is invoked the security context is already established 

418 and the client and server can send GSSAPI formated secure messages. 

419 

420 To finish the authentication handshake the server sends a message 

421 with the security layer availability and the maximum buffer size. 

422 

423 Since the connector only uses the GSSAPI authentication mechanism to 

424 authenticate the user with the server, the server will verify clients 

425 message signature and terminate the GSSAPI authentication and send two 

426 messages; an authentication acceptance b'\x01\x00\x00\x08\x01' and a 

427 OK packet (that must be received after sent the returned message from 

428 this method). 

429 

430 Args: 

431 message a wrapped hssapi message from the server. 

432 

433 Returns: bytearray closing handshake message to be send to the server. 

434 """ 

435 if not self.ctx.complete: 

436 raise errors.ProgrammingError("Security context is not completed.") 

437 _LOGGER.debug("# servers message: %s", message) 

438 _LOGGER.debug("# GSSAPI flags in use: %s", self.ctx.actual_flags) 

439 try: 

440 unwraped = self.ctx.unwrap(message) 

441 _LOGGER.debug("# unwraped: %s", unwraped) 

442 except gssapi.raw.exceptions.BadMICError as err: 

443 _LOGGER.debug("Unable to unwrap server message: %s", err) 

444 raise errors.InterfaceError("Unable to unwrap server message: {}" 

445 "".format(err)) 

446 

447 _LOGGER.debug("# unwrapped server message: %s", unwraped) 

448 # The message contents for the clients closing message: 

449 # - security level 1 byte, must be always 1. 

450 # - conciliated buffer size 3 bytes, without importance as no 

451 # further GSSAPI messages will be sends. 

452 response = bytearray(b"\x01\x00\x00\00") 

453 # Closing handshake must not be encrypted. 

454 _LOGGER.debug("# message response: %s", response) 

455 wraped = self.ctx.wrap(response, encrypt=False) 

456 _LOGGER.debug("# wrapped message response: %s, length: %d", 

457 wraped[0], len(wraped[0])) 

458 

459 return wraped.message

489 def _second_message(self): 

490 """This method generates the second message to the server 

491 

492 Second message consist on the concatenation of the client and the 

493 server nonce, and cproof. 

494 

495 c=<n,a=<user_name>>,r=<server_nonce>,p=<client_proof> 

496 where: 

497 <client_proof>: xor(<client_key>, <client_signature>) 

498 

499 <client_key>: hmac(salted_password, b"Client Key") 

500 <client_signature>: hmac(<stored_key>, <auth_msg>) 

501 <stored_key>: h(<client_key>) 

502 <auth_msg>: <client_first_no_header>,<servers_first>, 

503 c=<client_header>,r=<server_nonce> 

504 <client_first_no_header>: n=<username>r=<client_nonce> 

505 """ 

506 if not self._auth_data: 

507 raise errors.InterfaceError("Missing authentication data (seed)") 

508 

509 passw = self._normalize(self._password) 

510 salted_password = self._hi(passw, 

511 b64decode(self.server_salt), 

512 self.iterations) 

513 

514 _LOGGER.debug("salted_password: %s", 

515 b64encode(salted_password).decode()) 

516 

517 client_key = self._hmac(salted_password, b"Client Key") 

518 _LOGGER.debug("client_key: %s", b64encode(client_key).decode()) 

519 

520 stored_key = self.def_digest_mode(client_key).digest() 

521 _LOGGER.debug("stored_key: %s", b64encode(stored_key).decode()) 

522 

523 server_key = self._hmac(salted_password, b"Server Key") 

524 _LOGGER.debug("server_key: %s", b64encode(server_key).decode()) 

525 

526 client_first_no_header = ",".join([ 

527 "n={}".format(self._normalize(self._username)), 

528 "r={}".format(self.client_nonce)]) 

529 _LOGGER.debug("client_first_no_header: %s", client_first_no_header) 

530 auth_msg = ','.join([ 

531 client_first_no_header, 

532 self.servers_first, 

533 "c={}".format(b64encode("n,a={},".format( 

534 self._normalize(self._username)).encode()).decode()), 

535 "r={}".format(self.server_nonce)]) 

536 _LOGGER.debug("auth_msg: %s", auth_msg) 

537 

538 client_signature = self._hmac(stored_key, auth_msg.encode()) 

539 _LOGGER.debug("client_signature: %s", 

540 b64encode(client_signature).decode()) 

541 

542 client_proof = self._xor(client_key, client_signature) 

543 _LOGGER.debug("client_proof: %s", b64encode(client_proof).decode()) 

544 

545 self.server_auth_var = b64encode( 

546 self._hmac(server_key, auth_msg.encode())).decode() 

547 _LOGGER.debug("server_auth_var: %s", self.server_auth_var) 

548 

549 client_header = b64encode( 

550 "n,a={},".format(self._normalize(self._username)).encode()).decode() 

551 msg = ",".join(["c={}".format(client_header), 

552 "r={}".format(self.server_nonce), 

553 "p={}".format(b64encode(client_proof).decode())]) 

554 _LOGGER.debug("second_message: %s", msg) 

555 return msg.encode()

578 "".format(servers_first))

605 def _validate_second_reponse(self, servers_second): 

606 """Validates second message from the server. 

607 

608 The client and the server prove to each other they have the same Auth 

609 variable. 

610 

611 The second message from the server consist of the server's proof: 

612 server_proof = HMAC(<server_key>, <auth_msg>) 

613 where: 

614 <server_key>: hmac(<salted_password>, b"Server Key") 

615 <auth_msg>: <client_first_no_header>,<servers_first>, 

616 c=<client_header>,r=<server_nonce> 

617 

618 Our server_proof must be equal to the Auth variable send on this second 

619 response. 

620 """ 

621 if not servers_second or not isinstance(servers_second, bytearray) or \ 

622 len(servers_second)<=2 or not servers_second.startswith(b"v="):
